// Part 2: CSV to JSON // credit: Stack Overflow User "Paath" on 2016-05-07 // initialize variables let arrayOne = rawData.split("\r\n"); let header = arrayOne[0].split(","); let noOfRow = arrayOne.length; let noOfCol = header.length; let jsonData = []; let i = 0; let j = 0; // for loop (rows) for (i = 1; i < noOfRow - 1; i++) { let obj = {}; let myNewLine = arrayOne[i].split(","); // nested for loop (columns) for (j = 0; j < noOfCol; j++) { obj[header[j]] = myNewLine[j]; }; // generate JSON jsonData.push(obj); }; // publish data document.getElementById("json").innerHTML = jsonData; // developer info console.log(jsonData); console.table(jsonData); console.log("jsonData type: " + typeof jsonData); // Part 3: JSON to HTML Table // credit: Stack Overflow User "ray hatfield" on 2015-05-26 // initialize variables let children = jsonData; cell.appendChild(document.createTextNode(keys[i])); } } // generate table for (i = 0; i < children.length; i++) { let child = children[i]; // generate header row if (i === 0) { addHeaders(table, Object.keys(child));
